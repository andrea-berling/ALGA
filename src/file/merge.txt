function merge (A[], left, center, right)
	integer h
    integer i ← left
    integer j ← center + 1
    integer k ← left

    while i ≤ center and j ≤ right do
       if A[i] ≤ A[j] then
          B[k] ← A[i]		
          % B is a side array; can be global or passed as a parameter
          i ← i + 1
       else
          B[k] ← A[j]
          j ← j + 1
       k ← k + 1
    end while
    
    j ← right
    for h ← center downto i do
    	A[j] ← A[h]
    	j ← j - 1

    for j ← left to k-1 do
       A[j] ← B[j]
